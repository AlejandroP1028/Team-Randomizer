export function TeamRandomizerApp() {
  // Keyboard shortcuts
  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      const meta = e.metaKey || e.ctrlKey;
      if (meta && e.key === "z") { e.preventDefault(); useAppStore.getState().undo(); }
    };
    window.addEventListener("keydown", handler);
    return () => window.removeEventListener("keydown", handler);
  }, []);

    <div className="flex h-screen overflow-hidden bg-background">
      <aside className="w-[300px] shrink-0 border-r border-border flex flex-col overflow-y-auto">
        <InputPanel />
      </aside>
      <main className="flex-1 overflow-y-auto">
        <OutputPanel />
      </main>
    </div>
  );

// ─── components/InputPanel/ParticipantInput.tsx ───────────────────────────────
"use client";

import { useCallback } from "react";
import { useAppStore } from "@/store/useAppStore";
import type { Participant } from "@/lib/types";

function parseLines(raw: string): Participant[] {
  return raw
    .split("\n")
    .map(l => l.trim())
    .filter(Boolean)
    .map((name, i) => ({ id: `p_${i}_${name.toLowerCase().replace(/\s+/g, "_")}`, name }));
}

export function ParticipantInput() {
  const { participants, setParticipants } = useAppStore(s => ({
    participants: s.participants,
    setParticipants: s.setParticipants,
  }));

  const onChange = useCallback((e: React.ChangeEvent<HTMLTextAreaElement>) => {
    setParticipants(parseLines(e.target.value));
  }, [setParticipants]);

  const rawValue = participants.map(p => p.name).join("\n");

  return (
    <div className="flex flex-col gap-1.5">
      <label className="text-[11px] font-medium uppercase tracking-wide text-muted-foreground">
        Participants
      </label>
      <textarea
        value={rawValue}
        onChange={onChange}
        rows={8}
        placeholder={"Alice\nBob\nCarol\nDave\nEve\nFrank"}
        className="w-full rounded-md border border-border bg-background px-3 py-2
          text-sm leading-relaxed resize-none focus:outline-none focus:ring-1 focus:ring-ring"
      />
      <p className="text-xs text-muted-foreground">{participants.length} participants</p>
    </div>
  );
}


// ─── components/InputPanel/ConstraintConfig.tsx ───────────────────────────────
"use client";

import { useAppStore } from "@/store/useAppStore";
import type { Strategy } from "@/lib/types";

const STRATEGIES: { value: Strategy; label: string }[] = [
  { value: "balanced_skill",    label: "Balanced skill" },
  { value: "mixed_department",  label: "Mixed department" },
  { value: "random",            label: "Random" },
  { value: "custom",            label: "Custom" },
];

export function ConstraintConfig() {
  const { config, setConfig, participants } = useAppStore(s => ({
    config: s.config,
    setConfig: s.setConfig,
    participants: s.participants,
  }));

  const mode = config.teamCount != null ? "count" : "size";
  const numVal = mode === "count" ? config.teamCount ?? 3 : config.teamSize ?? 2;
  const maxTeams = Math.max(2, Math.floor(participants.length / 2));

  function setMode(m: "count" | "size") {
    if (m === "count") setConfig({ teamCount: numVal, teamSize: undefined });
    else               setConfig({ teamSize: numVal, teamCount: undefined });
  }

  function setNum(n: number) {
    if (mode === "count") setConfig({ teamCount: n });
    else                  setConfig({ teamSize: n });
  }

  return (
    <div className="flex flex-col gap-3">
      <div>
        <label className="text-[11px] font-medium uppercase tracking-wide text-muted-foreground block mb-1.5">
          Team size
        </label>
        <div className="flex gap-2 items-center">
          <div className="flex border border-border rounded-md overflow-hidden text-xs">
            {(["count", "size"] as const).map(m => (
              <button
                key={m}
                onClick={() => setMode(m)}
                className={`px-3 py-1.5 capitalize transition-colors
                  ${mode === m ? "bg-blue-50 text-blue-700 font-medium" : "text-muted-foreground hover:bg-muted"}`}
              >
                {m === "count" ? "Count" : "Size"}
              </button>
            ))}
          </div>
          <input
            type="number"
            value={numVal}
            min={2}
            max={mode === "count" ? maxTeams : Math.floor(participants.length / 2)}
            onChange={e => setNum(Math.max(2, parseInt(e.target.value) || 2))}
            className="w-14 rounded-md border border-border bg-background px-2 py-1 text-sm text-center
              focus:outline-none focus:ring-1 focus:ring-ring"
          />
          <span className="text-xs text-muted-foreground">
            {mode === "count" ? "teams" : "per team"}
          </span>
        </div>
      </div>

      <div>
        <label className="text-[11px] font-medium uppercase tracking-wide text-muted-foreground block mb-1.5">
          Strategy
        </label>
        <select
          value={config.strategy}
          onChange={e => setConfig({ strategy: e.target.value as Strategy })}
          className="w-full rounded-md border border-border bg-background px-2 py-1.5 text-sm
            focus:outline-none focus:ring-1 focus:ring-ring"
        >
          {STRATEGIES.map(s => <option key={s.value} value={s.value}>{s.label}</option>)}
        </select>
      </div>
    </div>
  );
}


// ─── components/InputPanel/PresetSidebar.tsx ─────────────────────────────────
"use client";

import { useState } from "react";
import { useAppStore } from "@/store/useAppStore";
import { useGenerate } from "@/hooks/useGenerate";

export function PresetSidebar() {
  const { presets, activePresetId, loadPreset, savePreset, deletePreset } = useAppStore(s => ({
    presets: s.presets,
    activePresetId: s.activePresetId,
    loadPreset: s.loadPreset,
    savePreset: s.savePreset,
    deletePreset: s.deletePreset,
  }));
  const { generate } = useGenerate();
  const [saving, setSaving] = useState(false);
  const [newName, setNewName] = useState("");

  function handleLoad(id: string) {
    loadPreset(id);
    generate();
  }

  function commitSave() {
    if (!newName.trim()) return;
    savePreset(newName.trim());
    setNewName(""); setSaving(false);
  }

  return (
    <div>
      <label className="text-[11px] font-medium uppercase tracking-wide text-muted-foreground block mb-1.5">
        Presets
      </label>
      <div className="flex flex-wrap gap-1.5">
        {presets.map(p => (
          <div key={p.id} className="group relative">
            <button
              onClick={() => handleLoad(p.id)}
              className={`px-2.5 py-1 text-xs rounded border transition-colors
                ${activePresetId === p.id
                  ? "border-blue-300 bg-blue-50 text-blue-700"
                  : "border-border text-muted-foreground hover:border-border-secondary"}`}
            >
              {p.name.length > 14 ? p.name.slice(0, 12) + "…" : p.name}
            </button>
            <button
              onClick={() => deletePreset(p.id)}
              className="absolute -top-1.5 -right-1.5 hidden group-hover:flex w-4 h-4 items-center
                justify-center rounded-full bg-muted border border-border text-[10px] text-muted-foreground hover:text-foreground"
            >
              ×
            </button>
          </div>
        ))}

        {saving ? (
          <input
            autoFocus
            value={newName}
            onChange={e => setNewName(e.target.value)}
            onKeyDown={e => { if (e.key === "Enter") commitSave(); if (e.key === "Escape") setSaving(false); }}
            onBlur={commitSave}
            placeholder="Preset name…"
            className="px-2 py-1 text-xs rounded border border-blue-300 focus:outline-none w-28"
          />
        ) : (
          <button
            onClick={() => setSaving(true)}
            className="px-2.5 py-1 text-xs rounded border border-dashed border-border
              text-blue-600 hover:border-blue-300 transition-colors"
          >
            + Save
          </button>
        )}
      </div>
    </div>
  );
}

"use client";

import { useAppStore } from "@/store/useAppStore";
import { TeamGrid } from "./TeamGrid";

import { ExportBar } from "./ExportBar";

  const { teams, warnings, undo, teamHistory } = useAppStore(s => ({
    teams: s.teams,
    warnings: s.warnings,
    undo: s.undo,
    teamHistory: s.teamHistory,
  }));

  const globalAvg = teams.length
    ? teams.flatMap(t => t.members).reduce((s, p) => s + (p.skillLevel ?? 3), 0) /

    <div className="flex flex-col gap-4 p-4 h-full">
      {/* Top bar */}
      <div className="flex items-center justify-between flex-wrap gap-2">
        <div className="flex items-center gap-2">
          {teams.length > 0 && (
            <span className="text-sm font-medium text-foreground">
              {teams.length} teams · avg skill {globalAvg.toFixed(1)}
            </span>
          )}
          {softWarnings.length > 0 && (
            <span className="px-2 py-0.5 text-xs rounded bg-amber-50 border border-amber-200 text-amber-700">
              {softWarnings.length} soft constraint{softWarnings.length > 1 ? "s" : ""} unmet
            </span>
          )}
        </div>
        <div className="flex gap-2">
          <button
            onClick={undo}
            disabled={!teamHistory.length}
            className="px-3 py-1.5 text-xs rounded border border-border hover:bg-muted
              disabled:opacity-40 disabled:cursor-not-allowed transition-colors"
          >
            ↩ Undo
          </button>
          <ExportBar />

      {/* Teams */}
      {teams.length > 0 ? (
        <>
          <TeamGrid />

        <div className="flex-1 flex items-center justify-center text-sm text-muted-foreground">

        </div>
      )}
    </div>
  );

// ─── components/OutputPanel/TeamGrid.tsx ─────────────────────────────────────
"use client";

import {
  DndContext, DragEndEvent, DragOverlay, DragStartEvent,
  PointerSensor, useSensor, useSensors, closestCenter,
} from "@dnd-kit/core";
import { useState } from "react";
import { useAppStore } from "@/store/useAppStore";
import { TeamCard } from "./TeamCard";
import { ParticipantCard } from "./ParticipantCard";
import type { Participant } from "@/lib/types";

interface DragData { participant: Participant; teamIndex: number; memberIndex: number; }

export function TeamGrid() {
  const { teams, swap, move } = useAppStore(s => ({ teams: s.teams, swap: s.swap, move: s.move }));
  const [active, setActive] = useState<DragData | null>(null);
  const sensors = useSensors(useSensor(PointerSensor, { activationConstraint: { distance: 5 } }));

  function onDragStart(e: DragStartEvent) {
    setActive(e.active.data.current as DragData);
  }

  function onDragEnd(e: DragEndEvent) {
    setActive(null);
    const src = e.active.data.current as DragData;
    const dst = e.over?.data.current as DragData | { teamIndex: number; memberIndex?: number } | undefined;
    if (!dst) return;

    const { teamIndex: srcTi, memberIndex: srcMi } = src;
    const dstTi = dst.teamIndex;

    // Drop onto a ParticipantCard → swap
    if ("memberIndex" in dst && dst.memberIndex !== undefined) {
      const dstMi = dst.memberIndex;
      if (srcTi === dstTi && srcMi === dstMi) return;
      swap(srcTi, srcMi, dstTi, dstMi);
    } else {
      // Drop onto a TeamCard zone → move
      if (srcTi === dstTi) return;
      move(srcTi, srcMi, dstTi);
    }
  }

  return (
    <DndContext sensors={sensors} collisionDetection={closestCenter} onDragStart={onDragStart} onDragEnd={onDragEnd}>
      <div className="grid gap-3" style={{ gridTemplateColumns: "repeat(auto-fill, minmax(180px, 1fr))" }}>
        {teams.map((team, ti) => (
          <TeamCard key={team.id} team={team} teamIndex={ti} />
        ))}
      </div>
      <DragOverlay>
        {active && (
          <div className="opacity-90 rotate-1 scale-105 shadow-md">
            <ParticipantCard participant={active.participant} teamIndex={active.teamIndex} memberIndex={active.memberIndex} />
          </div>
        )}
      </DragOverlay>
    </DndContext>
  );
}


// ─── components/OutputPanel/TeamCard.tsx ─────────────────────────────────────
"use client";

import { useDroppable } from "@dnd-kit/core";
import { useAppStore } from "@/store/useAppStore";
import { ParticipantCard } from "./ParticipantCard";
import type { Team } from "@/lib/types";

interface Props { team: Team; teamIndex: number; }

export function TeamCard({ team, teamIndex }: Props) {
  const globalAvg = useAppStore(s =>
    s.teams.flatMap(t => t.members).reduce((sum, p) => sum + (p.skillLevel ?? 3), 0) /
    Math.max(1, s.teams.flatMap(t => t.members).length)
  );

  const { setNodeRef, isOver } = useDroppable({
    id: `team-${teamIndex}`,
    data: { teamIndex },
  });

  const pct = Math.round((team.stats.avgSkill / 5) * 100);
  const inBalance = Math.abs(team.stats.avgSkill - globalAvg) <= 0.5;

  return (
    <div
      ref={setNodeRef}
      className={`rounded-xl border flex flex-col overflow-hidden transition-colors
        ${isOver ? "border-blue-300 bg-blue-50/50" : "border-border bg-card"}`}
    >
      {/* Header */}
      <div className="px-3 pt-3 pb-2 border-b border-border">
        <div className="flex items-baseline justify-between">
          <span className="text-sm font-medium text-foreground">{team.name}</span>
          <span className="text-xs text-muted-foreground">avg {team.stats.avgSkill}</span>
        </div>
        <div className="text-[11px] text-muted-foreground mt-0.5">
          {Object.entries(team.stats.departments).map(([d, n]) => `${d} ${n}`).join(" · ")}
        </div>
        {/* Skill bar */}
        <div className="mt-2 h-1 rounded-full bg-muted overflow-hidden">
          <div
            className={`h-full rounded-full transition-all ${inBalance ? "bg-green-500" : "bg-amber-400"}`}
            style={{ width: `${pct}%` }}
          />
        </div>
      </div>

      {/* Members */}
      <div className="p-2 flex flex-col gap-0.5">
        {team.members.map((p, mi) => (
          <ParticipantCard key={p.id} participant={p} teamIndex={teamIndex} memberIndex={mi} />
        ))}
      </div>
    </div>
  );
}


// ─── components/OutputPanel/ParticipantCard.tsx ───────────────────────────────
"use client";

import { useDraggable, useDroppable } from "@dnd-kit/core";
import type { Participant } from "@/lib/types";

const SKILL_STYLE: Record<number, { bg: string; text: string }> = {
  1: { bg: "bg-red-100",    text: "text-red-800"   },
  2: { bg: "bg-orange-100", text: "text-orange-800" },
  3: { bg: "bg-amber-100",  text: "text-amber-800"  },
  4: { bg: "bg-teal-100",   text: "text-teal-800"   },
  5: { bg: "bg-blue-100",   text: "text-blue-800"   },
};

interface Props { participant: Participant; teamIndex: number; memberIndex: number; }

export function ParticipantCard({ participant, teamIndex, memberIndex }: Props) {
  const dragData = { participant, teamIndex, memberIndex };

  const { attributes, listeners, setNodeRef: setDragRef, isDragging } = useDraggable({
    id: `${participant.id}-${teamIndex}-${memberIndex}`,
    data: dragData,
  });

  const { setNodeRef: setDropRef, isOver } = useDroppable({
    id: `chip-${teamIndex}-${memberIndex}`,
    data: dragData,
  });

  const skill = participant.skillLevel ?? 3;
  const style = SKILL_STYLE[skill] ?? SKILL_STYLE[3];

  return (
    <div
      ref={node => { setDragRef(node); setDropRef(node); }}
      className={`flex items-center gap-1.5 px-2 py-1.5 rounded-md text-xs cursor-grab select-none
        transition-all border
        ${isDragging ? "opacity-40" : ""}
        ${isOver ? "border-blue-300 bg-blue-50" : "border-transparent hover:bg-muted hover:border-border"}`}
      {...attributes}
      {...listeners}
    >
      {/* Drag handle */}
      <div className="flex flex-col gap-[3px] opacity-30 shrink-0">
        <span className="block w-3 h-px bg-foreground rounded" />
        <span className="block w-3 h-px bg-foreground rounded" />
        <span className="block w-3 h-px bg-foreground rounded" />
      </div>

      <span className="flex-1 text-foreground">{participant.name}</span>

      <span className={`px-1.5 py-0.5 rounded text-[10px] font-medium ${style.bg} ${style.text}`}>
        {skill}
      </span>

      {participant.department && (
        <span className="text-muted-foreground truncate max-w-[48px]">
          {participant.department}
        </span>
      )}
    </div>
  );
}
